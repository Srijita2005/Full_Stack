const fs = require('fs');
const readline = require('readline');

const FILE_NAME = 'employees.json';

const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout
});

let employees = [];
if (fs.existsSync(FILE_NAME)) {
    const data = fs.readFileSync(FILE_NAME, 'utf8');
    employees = JSON.parse(data || '[]');
}

const saveData = () => {
    fs.writeFileSync(FILE_NAME, JSON.stringify(employees, null, 2));
};

const showMenu = () => {
    console.log('\n--- Employee Management System ---');
    console.log('1. Add Employee');
    console.log('2. View All Employees');
    console.log('3. Update Employee');
    console.log('4. Delete Employee');
    console.log('5. Exit');
    rl.question('Select an option: ', handleInput);
};

const handleInput = (option) => {
    switch (option.trim()) {
        case '1':
            rl.question('Enter Name: ', (name) => {
                rl.question('Enter ID: ', (id) => {
                    rl.question('Enter Department: ', (dept) => {
                        if (!name || !id || !dept) {
                            console.log('Error: All fields are required.');
                        } else if (employees.find(e => e.id === id)) {
                            console.log('Error: Employee ID already exists.');
                        } else {
                            employees.push({ id, name, dept });
                            saveData();
                            console.log('Employee added successfully.');
                        }
                        showMenu();
                    });
                });
            });
            break;
        case '2':
            console.log('\n--- Employee List ---');
            if (employees.length === 0) console.log('No records found.');
            else console.table(employees);
            showMenu();
            break;
        case '3':
            rl.question('Enter Employee ID to Update: ', (id) => {
                const emp = employees.find(e => e.id === id);
                if (!emp) {
                    console.log('Employee not found.');
                    showMenu();
                } else {
                    rl.question(`Enter New Name (${emp.name}): `, (name) => {
                        rl.question(`Enter New Dept (${emp.dept}): `, (dept) => {
                            emp.name = name || emp.name;
                            emp.dept = dept || emp.dept;
                            saveData();
                            console.log('Employee updated successfully.');
                            showMenu();
                        });
                    });
                }
            });
            break;
        case '4':
            rl.question('Enter Employee ID to Delete: ', (id) => {
                const initialLength = employees.length;
                employees = employees.filter(e => e.id !== id);
                if (employees.length < initialLength) {
                    saveData();
                    console.log('Employee deleted successfully.');
                } else {
                    console.log('Employee not found.');
                }
                showMenu();
            });
            break;
        case '5':
            console.log('Exiting...');
            rl.close();
            break;
        default:
            console.log('Invalid option. Please try again.');
            showMenu();
            break;
    }
};

showMenu();
