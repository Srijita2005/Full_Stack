const express = require('express');
const app = express();
app.use(express.json());

let cards = [
    { id: 1, name: "Ace of Spades", suit: "Spades", value: "Ace", rarity: "Common" }
];

// GET All
app.get('/cards', (req, res) => res.json({ success: true, data: cards }));

// GET One
app.get('/cards/:id', (req, res) => {
    const card = cards.find(c => c.id == req.params.id);
    card ? res.json({ success: true, data: card }) : res.status(404).json({ error: "Not found" });
});

// POST
app.post('/cards', (req, res) => {
    const newCard = { id: cards.length + 1, ...req.body };
    cards.push(newCard);
    res.status(201).json({ success: true, data: newCard });
});

// PUT
app.put('/cards/:id', (req, res) => {
    const card = cards.find(c => c.id == req.params.id);
    if(card) {
        Object.assign(card, req.body);
        res.json({ success: true, data: card });
    } else {
        res.status(404).json({ error: "Not found" });
    }
});

// DELETE
app.delete('/cards/:id', (req, res) => {
    const idx = cards.findIndex(c => c.id == req.params.id);
    if(idx !== -1) {
        cards.splice(idx, 1);
        res.json({ success: true, message: "Deleted" });
    } else {
        res.status(404).json({ error: "Not found" });
    }
});

app.listen(3000, () => console.log('Server running on port 3000'));
