const express = require('express');
const { createClient } = require('redis');
const app = express();
const client = createClient();

app.use(express.json());

const TOTAL_SEATS = 20;
// In a real app, booked seats would be in a DB
let bookedSeats = new Set();

client.on('error', (err) => console.log('Redis Client Error', err));

(async () => {
    await client.connect();
    console.log("Connected to Redis");
})();

// Book Seat Endpoint
app.post('/book', async (req, res) => {
    const { seatId, userId } = req.body;

    if (seatId < 1 || seatId > TOTAL_SEATS) {
        return res.status(400).json({ error: "Invalid seat ID" });
    }

    if (bookedSeats.has(seatId)) {
        return res.status(400).json({ error: "Seat already booked" });
    }

    const lockKey = `lock:seat:${seatId}`;

    // ATTEMPT LOCK: Set key only if it doesn't exist (NX), expire in 5s (EX)
    const acquiredLock = await client.set(lockKey, userId, {
        NX: true,
        EX: 5
    });

    if (!acquiredLock) {
        return res.status(409).json({ error: "Seat is currently locked by another user. Try again later." });
    }

    // SIMULATE PAYMENT PROCESSING
    setTimeout(async () => {
        // Finalize booking
        bookedSeats.add(seatId);
        // Release lock (optional since it expires, but good practice if done early)
        await client.del(lockKey);
        console.log(`Seat ${seatId} successfully booked by ${userId}`);
    }, 2000);

    res.json({ status: "Processing", message: "Seat locked. Payment processing..." });
});

app.listen(3000, () => console.log('Server running on port 3000'));
